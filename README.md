## Installation

```bash
npm i animapi
```

### Usage

```ts
import { Otakudesu } from "animapi";

const otakudesu = new Otakudesu();

/**
 * @return
 * {
 *    hasNext: true
 *    animes: [...]
 * }
 */
const res = await otakudesu.search({
  filter: {
    keyword: "kimetsu",
  },
});
```
